--6. Cree 3 perfiles:
--a. Perfil 1: "Manager": Tiempo de una sesión por usuario, 40 días de duración de la contraseña, 15 mins inactivo, 4 intentos fallidos para inicio de sesión.

CREATE PROFILE MANAGER LIMIT
SESSIONS_PER_USER 1
CONNECT_TIME 15
PASSWORD_LIFE_TIME 40  
FAILED_LOGIN_ATTEMPTS 4;


--b. Perfil 2: "finance": Tiempo de una sesión por usuario, 15 días de duración de la contraseña, 3 mins inactivo, 2 intentos fallidos para inicio de sesión.

CREATE PROFILE FINANCE LIMIT
SESSIONS_PER_USER 1
CONNECT_TIME 3
PASSWORD_LIFE_TIME 15
FAILED_LOGIN_ATTEMPTS 2;


--c. Perfil 2: "development": Tiempo de dos sesión por usuario, 100 días de duración de la contraseña, 30 mins inactivo, 0 intentos fallidos para inicio de sesión.

CREATE PROFILE DEVELOPMENT LIMIT
SESSIONS_PER_USER 2
CONNECT_TIME 30
PASSWORD_LIFE_TIME 100;

 CREATE TABLE FABRICANTES
( ID NUMBER(10) GENERATED AS IDENTITY,
  IDFABRICANTE NUMBER(10) NOT NULL,
  DESCRIPCION VARCHAR2(50) NOT NULL,
  CONSTRAINT FABRICANTE_PK PRIMARY KEY (IDFABRICANTE));
  
  
 CREATE TABLE MARCAS
( ID NUMBER(10) GENERATED AS IDENTITY,
  IDMARCA NUMBER(10) NOT NULL,
  DESCRIPCION VARCHAR2(50) NOT NULL,
  CONSTRAINT MARCA_PK PRIMARY KEY (IDMARCA));

CREATE TABLE TIPOVEHICULOS
( ID NUMBER(10) GENERATED AS IDENTITY,
  IDTIPO NUMBER(1) NOT NULL,
  DESCRIPCION VARCHAR2(20),
  CONSTRAINT TIPOVEH_PK PRIMARY KEY (IDTIPO));

 CREATE TABLE VEHICULOS
( ID NUMBER(10) GENERATED AS IDENTITY,
  VIN NUMBER(10) NOT NULL,
  IDMARCA NUMERIC(5) NOT NULL,
  MODELO NUMERIC(4),
  IDFABRICA NUMERIC(11) NOT NULL,
  IDTIPOVEH NUMBER(1) NOT NULL,
  CONSTRAINT VEHICULOS_PK PRIMARY KEY (VIN),
  CONSTRAINT VEHI_FABRI_FK FOREIGN KEY (IDFABRICA) REFERENCES FABRICANTES(IDFABRICANTE),
  CONSTRAINT VEHI_MARCA_FK FOREIGN KEY (IDMARCA) REFERENCES MARCAS(IDMARCA),
  CONSTRAINT VEHI_TIPO_FK FOREIGN KEY (IDTIPOVEH) REFERENCES TIPOVEHICULOS(IDTIPO)
  );
  
  CREATE TABLE CLIENTES
( ID NUMBER(10) GENERATED AS IDENTITY,
  IDENTIFICACION NUMBER(10) NOT NULL,
  NOMBRE VARCHAR2(50) NOT NULL,
  APELLIDO VARCHAR2(50) NOT NULL,
  DIRECCION  VARCHAR2(50) NOT NULL,
  EMAIL VARCHAR2(40) ,
  TELEFONO VARCHAR2(20),
  CONSTRAINT CLIENTES_PK PRIMARY KEY (IDENTIFICACION)
);

CREATE TABLE TIPOVENDEDOR
( ID NUMBER(10) GENERATED AS IDENTITY,
  IDTIPO NUMBER(1) NOT NULL,
  DESCRIPCION VARCHAR2(20),
  CONSTRAINT TIPOVENDE_PK PRIMARY KEY (IDTIPO));
  
  
  
  CREATE TABLE VENDEDORES
( ID NUMBER(10) GENERATED AS IDENTITY,
  IDENTIFICACION NUMBER(10) NOT NULL,
  NOMBRE VARCHAR2(50) NOT NULL,
  APELLIDO VARCHAR2(50) NOT NULL,
  DIRECCION  VARCHAR2(50) NOT NULL,
  EMAIL VARCHAR2(40) ,
  TELEFONO VARCHAR2(20),
  IDTIPOVENDEDOR NUMBER(10)  NOT NULL,
  CONSTRAINT VENDEDOR_PK PRIMARY KEY (IDENTIFICACION),
  CONSTRAINT VENDE_TIPO_FK FOREIGN KEY (IDTIPOVENDEDOR) REFERENCES TIPOVENDEDOR (IDTIPO)
);





CREATE TABLE PARTEVEHICULOS
( ID NUMBER(10) GENERATED AS IDENTITY,
  CODPARTE NUMBER(10) NOT NULL,
  DESCRIPCION VARCHAR2(50) NOT NULL,
  VLRCOSTO NUMBER(10) NOT NULL,
  VLRPRECIO NUMBER(12) NOT NULL,
  CONSTRAINT PARTEVEH_PK PRIMARY KEY (CODPARTE));
  
  
 CREATE TABLE FACTURAS
 ( ID NUMBER(10) GENERATED AS IDENTITY,
  NROFACTURA NUMBER(10) NOT NULL,
  IDCLILENTE NUMBER(10) NOT NULL,
  IDVENDEDOR NUMBER(10) NOT NULL,
  FECHA DATE NOT NULL,
  SUBTOTAL NUMBER(12) NOT NULL,
  IMPUESTOS NUMBER (12) NOT NULL,
  TOTAL NUMBER (12) NOT NULL,
  CONSTRAINT FACTURA_PK PRIMARY KEY (NROFACTURA),
  CONSTRAINT FAC_CLIENTE_FK FOREIGN KEY (IDCLILENTE) REFERENCES CLIENTES(IDENTIFICACION),
  CONSTRAINT FAC_VENDEDOR_FK FOREIGN KEY (IDVENDEDOR) REFERENCES VENDEDORES(IDENTIFICACION)
  );
 
 CREATE TABLE DETALLEFACTURAS
 (ID NUMBER(10) GENERATED AS IDENTITY,
  NRODETALFACTURA NUMBER(10) NOT NULL,
  NROFACTURA NUMBER(10) NOT NULL,
  CODPARTE NUMBER(10) NOT NULL,
  VINNUEVO NUMBER(10) NOT NULL,
  VINUSADO NUMBER(10),
  VLRPARTE NUMBER(10) NOT NULL,
  CONSTRAINT DETFACT_PK PRIMARY KEY (NRODETALFACTURA),
  CONSTRAINT DETFAC_FK FOREIGN KEY (NROFACTURA) REFERENCES FACTURAS(NROFACTURA),
  CONSTRAINT DETFAC_PARTE_FK FOREIGN KEY (CODPARTE) REFERENCES PARTEVEHICULOS (CODPARTE) 
 );


 CREATE TABLE VEHICULOINTERNOVENTA
 ( ID NUMBER(10) GENERATED AS IDENTITY,
  NROVEHICULOVENTA NUMBER(10) NOT NULL,
  VINNUEVO NUMBER(10) NOT NULL,
  VINUSADO NUMBER(10) NOT NULL,
  FECHACAMBIO DATE NOT NULL,
  IDCLILENTE NUMBER(10) NOT NULL,
  NROFACTURA NUMBER(10) NOT NULL,
  CONSTRAINT VEHICULOINTERNOVENTA_PK PRIMARY KEY (NROVEHICULOVENTA),
  CONSTRAINT VEHIVENTA_FACTURA_FK FOREIGN KEY (NROFACTURA) REFERENCES FACTURAS(NROFACTURA),
  CONSTRAINT VEHIVENTA_CLIENTE_FK FOREIGN KEY (IDCLILENTE) REFERENCES CLIENTES(IDENTIFICACION),  
  CONSTRAINT VEHIVENTA_VEHICULO_FK FOREIGN KEY (VINNUEVO) REFERENCES VEHICULOS(VIN));
